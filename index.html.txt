<script>
let startTime=0;
let elapsed=0;
let timer=null;
let slider=document.getElementById("unitSlider");
let label=document.getElementById("unitLabel");

slider.oninput=function(){
    label.innerText=this.value;
    // タイマーが停止している時でも、スライダーを動かしたら特殊時間の表示を更新する
    if(!timer) {
        render();
    }
}

function start(){
    if(timer)return;
    startTime=Date.now()-elapsed;
    timer=setInterval(update,10);
}

function stop(){
    clearInterval(timer);
    timer=null;
}

function reset(){
    clearInterval(timer);
    timer=null;
    elapsed=0;
    document.getElementById("normal").innerText="0.00";
    document.getElementById("special").innerText="0'00";
}

function update(){
    elapsed=Date.now()-startTime;
    render();
}

// 画面への描画ロジックを共通化
function render(){
    let seconds=elapsed/1000;
    document.getElementById("normal").innerText=seconds.toFixed(2);
    
    let unit=parseFloat(slider.value);
    let specialSeconds=seconds/unit;
    let main=Math.floor(specialSeconds);
    let fraction=specialSeconds-main;
    
    // 計算誤差を防ぐため、Math.round（四捨五入）に変更
    let milli=Math.round(fraction*100); 
    if(milli >= 100) { 
        milli = 0;
        main += 1;
    }
    
    let milliStr=milli.toString().padStart(2,"0");
    document.getElementById("special").innerText=main+"'"+milliStr;
}
</script>
